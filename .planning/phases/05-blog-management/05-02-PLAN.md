```html
{% extends "admin/base.html" %}

{% block admin_title %}New Article{% endblock %}

{% block admin_content %}
<div class="editor-container">
    <form method="POST" action="{{ url_for('admin.create') }}" class="article-form">
        <input type="hidden" name="csrf_token" value="{{ csrf_token }}">
        
        <div class="form-group">
            <label for="title">Title</label>
            <input type="text" id="title" name="title"

        </div>

            <label for="status">Status</label>
            <select id="status" name="status">
                <option value="draft" {% if form_data.get('status') == 'draft' %}selected{% endif %}>Draft</option>
                <option value="published" {% if form_data.get('status') == 'published' %}selected{% endif %}>Published</option>
            </select>
        </div>
        
        <div class="editor-layout">
            <div class="editor-pane">
                <label for="content">Content (Markdown)</label>
                <textarea id="content" name="content" 
                          placeholder="Write your article content in Markdown..."
                          oninput="updatePreview()">{{ form_data.get('content', '') if form_data else '' }}</textarea>
            </div>
            
            <div class="preview-pane">
                <label>Preview</label>
                <div id="preview" class="markdown-preview">
                    <p class="placeholder-text">Start typing to see preview...</p>
                </div>
            </div>
        </div>
        
        <div class="form-actions">
            <button type="submit" class="btn btn-primary">Create Article</button>
            <a href="{{ url_for('admin.list') }}" class="btn btn-secondary">Cancel</a>
        </div>
    </form>
</div>
{% endblock %}

{% block extra_js %}
<script>
// Preview update function with debouncing
let debounceTimer;
function updatePreview() {
    clearTimeout(debounceTimer);
    debounceTimer = setTimeout(() => {
        const content = document.getElementById('content').value;
        const preview = document.getElementById('preview');
        
        if (content.trim()) {
            // Send to server for rendering
            fetch('/admin/preview', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/x-www-form-urlencoded',
                },
                body: 'content=' + encodeURIComponent(content)
            })
            .then(response => response.text())
            .then(html => {
                preview.innerHTML = html;
            })
            .catch(err => {
                preview.innerHTML = '<p class="error">Preview error</p>';
            });
        } else {
            preview.innerHTML = '<p class="placeholder-text">Start typing to see preview...</p>';
        }
    }, 300);
}
</script>
{% endblock %}
```

3. Add preview route to `app/admin/routes.py`:

```python
@bp.route('/preview', methods=['POST'])
@login_required
def preview():
    """Render markdown preview for admin editor."""
    from flask import Markup
    from ..markdown import render_markdown
    
    content = request.form.get('content', '')
    if content:
        html = render_markdown(content)
        return html
    return '<p class="placeholder-text">Start typing to see preview...</p>'
```
